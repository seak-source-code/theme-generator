<script>

function generate() {

  const themeKey = document.getElementById("themeKey").value;

  const country = document.getElementById("country").value;

  const mode = document.getElementById("mode").value;


  if (!themeKey) {

    alert("Please enter themeKey");

    return;

  }


  const theme = themeKey.toLowerCase();

  const prefix = country === "kh" ? "abc" : "abl";


  const menu = (name) => `ic-${name}-${theme}-${country}-${mode}.png`;


  const data = {

    themeKey: themeKey,

    thumbnail: `${prefix}-${theme}.png`,

    backgroundImage: `bg-${theme}-${country}-${mode}.png`,

    balanceBgImage: `bgbl-${theme}-${country}-${mode}.png`,

    eventBalaneImageInside: `evbl-${theme}-${country}-${mode}.png`,

    appearanceImage: `appear-${theme}-${country}-${mode}.png`,

    qrScanImageButton: `qr-${theme}-${country}-${mode}.png`,

    mainMenu: {

      account: menu("acc"),

      payment: menu("payment"),

      topup: menu("topup"),

      card: menu("card"),

      scanqr: menu("scanqr"),

      transfer: menu("transfer"),

      deposit: menu("deposit"),

      loan: menu("loan"),

      faqs: menu("faqs"),

      quickcash: menu("quickcash")

    },

    extraMenu: {

      exgrate: menu("exgrate"),

      publicservice: menu("publicservice"),

      csx: menu("csx"),

      promotion: menu("promotion"),

      location: menu("location"),

      invitefri: menu("invitefri"),

      security: menu("security"),

      payme: menu("payme"),

      acsmry: menu("acsmry"),

      trade: menu("trade"),

      acpay: menu("acpay"),

      merchant: menu("merchant")

    }

  };


  document.getElementById("output").value =

    JSON.stringify(data, null, 2);

}


function download() {

  const content = document.getElementById("output").value;


  if (!content) {

    alert("Generate JSON first");

    return;

  }


  const blob = new Blob([content], { type: "application/json" });

  const a = document.createElement("a");

  a.href = URL.createObjectURL(blob);

  a.download = "theme.json";

  a.click();

}
